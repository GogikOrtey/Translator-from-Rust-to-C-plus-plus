```c++
#include<iostream>
#include<string>
using namespace std;

int main() {
    int a = 5; // int
    float b = 4.5; // float
    bool c = true; // bool
    char d = 'R'; // char
    string e = "Hello, C++!"; // string

    // Условный оператор
    if (a == 5) {
        cout << "a равно 5" << endl;
    } else {
        cout << "a не равно 5" << endl;
    }

    // Блок кода
    for (int i = 0; i < 10; i++) {
        cout << "i равно " << i << endl;
    }

    // Множественные операции
    a = (4 * a) - (24 * a);

    return 0;
}
```
