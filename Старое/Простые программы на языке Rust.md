#### Простые программы на языке Rust

Программа для вычисления факториала числа

~~~rust
// Функция для вычисления факториала числа
fn factorial(n: u64) -> u64 {
    match n {
        // Если n равно 0, то факториал равен 1
        0 => 1,
        // Иначе, умножаем n на факториал (n-1)
        _ => n * factorial(n - 1),
    }
}

fn main() {
    // Вызываем функцию factorial для числа 5 и выводим результат в консоль
    let n = 5;
    println!("The factorial of {} is {}", n, factorial(n));
}
~~~

Программа для вычисления суммы чисел от 1 до N

```rust
use std::io; // Подключаем стандартную библиотеку ввода-вывода

fn main() {
    println!("Введите число N:"); // Просим пользователя ввести число N

    let mut input = String::new(); // Создаем новую переменную для ввода

    io::stdin().read_line(&mut input).expect("Ошибка чтения строки"); // Читаем строку из входного потока

    let n: u32 = input.trim().parse().expect("Не удалось преобразовать в число"); // Преобразуем строку в число и сохраняем в переменной n

    let mut sum = 0; // Создаем переменную для хранения суммы

    for i in 1..=n { // Итерируемся от 1 до n включительно
        sum += i; // Добавляем текущее значение i к сумме
    }

    println!("Сумма чисел от 1 до {} равна {}", n, sum); // Выводим результат на экран
}
```

Программа для вычисления корней квадратного уравнения

```rust
use std::io; // Подключаем стандартную библиотеку ввода-вывода
use std::f64; // Подключаем стандартную библиотеку для работы с числами с плавающей точкой

fn main() {
    println!("Введите коэффициенты a, b и c:"); // Просим пользователя ввести коэффициенты

    let mut input = String::new(); // Создаем новую переменную для ввода

    io::stdin().read_line(&mut input).expect("Ошибка чтения строки"); // Читаем строку из входного потока

    let a: f64 = input.trim().parse().expect("Не удалось преобразовать в число"); // Преобразуем строку в число и сохраняем в переменной a

    input.clear(); // Очищаем переменную для ввода

    io::stdin().read_line(&mut input).expect("Ошибка чтения строки"); // Читаем строку из входного потока

    let b: f64 = input.trim().parse().expect("Не удалось преобразовать в число"); // Преобразуем строку в число и сохраняем в переменной b

    input.clear(); // Очищаем переменную для ввода

    io::stdin().read_line(&mut input).expect("Ошибка чтения строки"); // Читаем строку из входного потока

    let c: f64 = input.trim().parse().expect("Не удалось преобразовать в число"); // Преобразуем строку в число и сохраняем в переменной c

    let discriminant = b.powf(2.0) - 4.0 * a * c; // Вычисляем дискриминант

    if discriminant > 0.0 { // Если дискриминант больше нуля
        let x1 = (-b + f64::sqrt(discriminant)) / (2.0 * a); // Вычисляем первый корень
        let x2 = (-b - f64::sqrt(discriminant)) / (2.0 * a); // Вычисляем второй корень

        println!("Корни уравнения: x1 = {}, x2 = {}", x1, x2); // Выводим результат на экран
    } else if discriminant == 0.0 { // Если дискриминант равен нулю
        let x = -b / (2.0 * a); // Вычисляем корень

        println!("Уравнение имеет единственный корень: x = {}", x); // Выводим результат на экран
    } else { // Если дискриминант меньше нуля
        println!("Уравнение не имеет действительных корней"); // Выводим сообщение об ошибке на экран
    }
}
```

Программа для вычисления среднего арифметического чисел

```rust
use std::io; // Подключаем стандартную библиотеку ввода-вывода

fn main() {
    println!("Введите числа через пробел:"); // Просим пользователя ввести числа

    let mut input = String::new(); // Создаем новую переменную для ввода

    io::stdin().read_line(&mut input).expect("Ошибка чтения строки"); // Читаем строку из входного потока

    let nums: Vec<f64> = input // Создаем новый вектор из чисел, разделенных пробелами
        .split_whitespace() // Разделяем строку на числа по пробелам
        .map(|s| s.parse().unwrap()) // Преобразуем каждое число в тип f64
        .collect(); // Собираем все числа в вектор

    let sum = nums.iter().sum::<f64>(); // Вычисляем сумму чисел в векторе

    let count = nums.len() as f64; // Получаем количество чисел в векторе и преобразуем в тип f64

    let average = sum / count; // Вычисляем среднее арифметическое

    println!("Среднее арифметическое: {}", average); // Выводим результат на экран
}
```

Программа для вычисления синуса и косинуса

~~~rust
use std::io; // Подключаем стандартную библиотеку ввода-вывода
use std::f64::consts::PI; // Подключаем константу PI из стандартной библиотеки для работы с числами с плавающей точкой

fn main() {
    println!("Введите угол в градусах:"); // Просим пользователя ввести угол

    let mut input = String::new(); // Создаем новую переменную для ввода

    io::stdin().read_line(&mut input).expect("Ошибка чтения строки"); // Читаем строку из входного потока

    let angle: f64 = input.trim().parse().expect("Не удалось преобразовать в число"); // Преобразуем строку в число и сохраняем в переменной angle

    let radians = angle * PI / 180.0; // Преобразуем угол из градусов в радианы

    let sin = f64::sin(radians); // Вычисляем синус угла

    let cos = f64::cos(radians); // Вычисляем косинус угла

    println!("Синус угла {} равен {}", angle, sin); // Выводим результат на экран

    println!("Косинус угла {} равен {}", angle, cos); // Выводим результат на экран
}
~~~
